#' Find Optimal Short-Form Scales
#'
#' @description
#' This function systematically evaluates subsets of items from a larger scale
#' to find the combination that correlates most highly with a target criterion.
#' It includes optimizations for handling a very large number of combinations,
#' such as iterative item reduction (optimisation) and memory-efficient batch processing.
#'
#' @param data A numeric matrix or data.frame containing the item responses.
#' @param n.items The desired number of items in the final short-form scale.
#' @param n.sets The number of top-performing item sets to return. Defaults to 5.
#' @param item.names Logical. If TRUE, the output will list the item names instead
#'   of their column numbers. Defaults to FALSE.
#' @param r.sq Logical. If TRUE, the squared correlation (R-squared) is returned
#'   alongside the correlation. Defaults to FALSE.
#' @param generate Logical. If TRUE, the function returns the computed scores for
#'   the single best item set. Defaults to FALSE.
#' @param item.set An integer specifying which item set to generate scores for
#'   (e.g., 1 for the best set). Only used if `generate = TRUE`.
#' @param na.rm Logical. If TRUE, missing values are handled by pairwise deletion
#'   or row-wise removal in means. Defaults to TRUE.
#' @param targ An optional numeric vector to serve as the target criterion.
#'   If NULL (the default), the mean score of the full scale is used.
#' @param optimise Logical or NULL. Controls the optimisation process.
#'   - `NULL` (default): If combinations exceed `ceiling`, prompts the user to
#'     choose between continuing, optimising, or quitting.
#'   - `TRUE`: Automatically runs the optimisation process if combinations
#'     exceed `ceiling`.
#'   - `FALSE`: Proceeds with the full, non-optimised process, even if
#'     combinations exceed `ceiling` (after a warning).
#' @param scale Logical. Defaults to FALSE, in which case mean-centered; if TRUE it will mean-centre and scale
#'   all columns in the dataset.
#' @param ceiling Numeric. The number of combinations above which the `optimise`
#'   prompt or process is triggered. Defaults to 100,000.
#'
#' @return A data.frame listing the top `n.sets` of items and their correlation
#'   with the target. If `generate = TRUE`, returns a numeric vector of scores
#'   for the selected `item.set`.
#'
#' @author Paddy Maher, Goldsmiths, University of London. Contact: paddycarstenmaher@gmail.com
#'
#' @examples
#' \dontrun{
#' library(MASS) 
#' # Note: The 'psych' package is required for pca() if using optimisation.
#' library(psych)
#' 
#' # 1. Create simulated data
#' set.seed(42)
#' loadings <- c(rnorm(10,0.75,0.08)*sign(rnorm(10)))
#' target_cor_matrix <- outer(loadings, loadings)
#' diag(target_cor_matrix) <- 1
#' simulated_data <- MASS::mvrnorm(n = 200, mu = rep(0, 10), 
#'                                 Sigma = target_cor_matrix, empirical = TRUE)
#' simulated_data <- as.data.frame(simulated_data)
#' colnames(simulated_data) <- paste0("item_", 1:10)
#' 
#' # 2. Reduce the scale to 3 items
#' reduceTo(simulated_data, 3)
#' 
#' # 3. Reduce a larger set, triggering the optimisation
#' large_data <- as.data.frame(matrix(rnorm(200*30), 200, 30))
#' colnames(large_data) <- paste0("item_", 1:30)
#' reduceTo(large_data, 5, optimise = TRUE, item.names = TRUE)
#' }
#'
#' @export

reduceTo <- function (data, n.items, n.sets = 5, item.names = FALSE, r.sq = FALSE, 
                      generate = FALSE, item.set = 1, na.rm = TRUE, targ = NULL, 
                      optimise = NULL, scale.vars = FALSE, ceiling = 10^5) 
{
  
  # --- Helper Function for Optimisation ---
  perform_optimisation <- function(current_cols, data, n.items, ceiling, targ) {
    
    # [PERF] No need to check is.numeric here; handled in main setup now.
    
    num_combs <- choose(length(current_cols), n.items)
    
    # Limit data to 20k rows during the optimisation process
    n_rows <- nrow(data)
    if (n_rows > 20000) {
      opt_indices <- sample(1:n_rows, 20000)
    } else {
      opt_indices <- 1:n_rows
    }
    
    if (is.null(targ)) {
      # PCA Strategy
      data_sample <- data[opt_indices, current_cols, drop = FALSE]
      
      while (num_combs > ceiling) {
        pca_res <- psych::pca(data_sample, nfactors = 1)
        local_rank <- order(abs(pca_res$loadings), decreasing = TRUE) 
        n_to_keep <- round(length(local_rank) * 0.9)
        top_local_indices <- local_rank[1:n_to_keep]
        
        current_cols <- current_cols[top_local_indices]
        data_sample <- data_sample[, top_local_indices, drop = FALSE]
        num_combs <- choose(length(current_cols), n.items)
      }
    } else {
      # Target Strategy
      cor_res <- cor(data[opt_indices, current_cols], 
                     targ[opt_indices], 
                     use = "pairwise.complete.obs")
      
      item_rank <- order(abs(as.vector(cor_res)), decreasing = TRUE) 
      
      k <- length(current_cols)
      while (choose(k, n.items) > ceiling && k > n.items) {
        k <- floor(k * 0.9) 
      }
      if(k < n.items) k <- n.items
      
      top_local_indices <- item_rank[1:k]
      current_cols <- current_cols[top_local_indices]
    }
    
    return(current_cols) 
  }
  
  # --- Initial Setup & Auto-Cleaning ---
  
  # 1. Save original names/indices before we touch anything
  if(is.null(colnames(data))) colnames(data) <- paste0("Col_", 1:ncol(data))
  all_original_indices <- 1:ncol(data)
  
  # 2. Identify Numeric Columns that are NOT all NA
  # Using sapply is efficient enough here even for large cols
  valid_mask <- sapply(data, function(x) is.numeric(x) && !all(is.na(x)))
  
  if (sum(valid_mask) < n.items) {
    stop("Error: Input data contains fewer valid numeric columns than 'n.items'.")
  }
  
  # 3. Filter the dataset
  # We overwrite 'data' with only the valid columns
  data <- data[, valid_mask]
  
  # 4. Update the Index Tracker
  # 'cols' starts as 1..N of the NEW valid data
  # 'original_indices' keeps track of where these came from in the raw file
  cols <- 1:ncol(data)
  original_indices <- all_original_indices[valid_mask]
  
  # 5. Convert to matrix for speed
  if(!is.matrix(data)) data <- as.matrix(data)
  
  # Check combinations based on the CLEANED data
  num_combinations <- choose(ncol(data), n.items)
  warning_msg <- paste("This will generate", format(num_combinations, big.mark = ","), 
                       "combinations.")
  
  # --- Handle Large Combinations & Optimisation ---
  if (num_combinations > ceiling) {
    if (is.null(optimise)) {
      message(warning_msg)
      user_choice <- readline(prompt = "Enter 'c' to continue, 'o' for optimisation, or 'q' to quit: ")
      if (tolower(user_choice) == "o") {
        cols <- perform_optimisation(cols, data, n.items, ceiling, targ)
      } else if (tolower(user_choice) != "c") {
        stop("Process aborted.")
      }
    } else if (optimise == TRUE) {
      message("Optimisation is enabled. Reducing item set...")
      cols <- perform_optimisation(cols, data, n.items, ceiling, targ)
    }
  }
  
  # Recalculate combinations
  num_combinations_final <- choose(length(cols), n.items)
  
  # --- Data Preparation ---
  
  # Update 'original_indices' to reflect only the OPTIMISED columns
  # 'cols' currently contains indices of the 'data' matrix (which is already filtered for numeric)
  original_indices <- original_indices[cols] 
  
  # Subset data to optimised set
  data <- data[, cols, drop = FALSE] 
  
  # Flip items
  if (is.null(targ)){
    dc <- cor(data, use = "pairwise.complete.obs")
    most_central_item <- order(colSums(abs(dc), na.rm = TRUE), decreasing = TRUE)[1]
    items_to_flip <- dc[, most_central_item] < 0
  } else {
    items_to_flip <- as.vector(cor(data, targ, use = "pairwise.complete.obs")) < 0
  }
  
  if(scale.vars) data <- apply(data, 2, scale)
  data[, items_to_flip] <- data[, items_to_flip] * -1
  
  # --- Batch Processing ---
  item_combinations <- combn(1:ncol(data), n.items)
  num_combinations <- ncol(item_combinations) 
  leaderboard <- data.frame(combination = character(0), r = numeric(0))
  batch_size <- 10000 
  num_batches <- ceiling(num_combinations / batch_size)
  
  if (is.null(targ)) targ <- rowMeans(data, na.rm = na.rm)
  
  for (j in 1:num_batches) {
    start_col <- (j - 1) * batch_size + 1
    end_col <- min(j * batch_size, num_combinations)
    current_batch_combs <- item_combinations[, start_col:end_col, drop = FALSE]
    
    scores_batch <- matrix(NA, ncol = ncol(current_batch_combs), nrow = nrow(data))
    for (i in 1:ncol(current_batch_combs)) {
      scores_batch[, i] <- rowMeans(data[, current_batch_combs[, i], drop = FALSE], na.rm = na.rm)
    }
    
    # Name the combinations
    if (item.names) {
      colnames(scores_batch) <- apply(current_batch_combs, 2, function(x) paste(colnames(data)[x], collapse = ","))
    } else {
      # Use the tracked original indices
      colnames(scores_batch) <- apply(current_batch_combs, 2, function(x) paste(original_indices[x], collapse = ","))    
    }
    
    cor_results_batch <- cor(scores_batch, targ, use = "pairwise")
    combined_results <- rbind(leaderboard, data.frame(combination = rownames(cor_results_batch),
                                                      r = cor_results_batch[,1]))
    leaderboard <- combined_results[order(abs(combined_results$r), decreasing = TRUE), ]
    leaderboard <- head(leaderboard, n.sets)
    rownames(leaderboard) <- 1:n.sets
    
    cat('\rProcessing Batch:', j, 'of', num_batches, 
        '| Total combinations:', format(num_combinations_final, big.mark = ","))
    flush.console()
  }
  
  cat('\r', rep(' ', 80), '\n', sep = '')
  best_set_string <- leaderboard$combination[item.set]
  best_set_indices <- as.numeric(strsplit(best_set_string, ',')[[1]])
  
  # --- Return Desired Output ---
  
  # 1. Helper logic to extract indices/names for the BEST set (Rank 1)
  #    We do this automatically so you always have the best indices saved.
  top_set_string <- rownames(leaderboard)[1] 
  top_indices_raw <- as.numeric(strsplit(top_set_string, ',')[[1]])
  
  best_names <- NULL
  best_indices <- NULL
  
  if (any(is.na(top_indices_raw))) {
    # Case: Item Names were used
    best_names <- strsplit(top_set_string, ',')[[1]]
    # Try to find indices if possible
    best_indices <- match(best_names, colnames(data)) 
  } else {
    # Case: Numeric Indices were used
    best_indices <- top_indices_raw
    # Find the corresponding names
    # Note: We need to map Original Indices -> Local Data Indices -> Names
    local_indices <- match(best_indices, original_indices)
    best_names <- colnames(data)[local_indices]
  }
  
  # 2. Generate Scores (if requested)
  #    If generate=TRUE, we calculate the scores and put them in the list.
  computed_scores <- NULL
  if (generate) {
    # Use the specific item.set requested by the user (default is 1)
    target_set_string <- rownames(leaderboard)[item.set]
    target_indices_raw <- as.numeric(strsplit(target_set_string, ',')[[1]])
    
    if (any(is.na(target_indices_raw))) {
      # Names input
      target_cols <- strsplit(target_set_string, ',')[[1]]
      computed_scores <- scale(rowMeans(data[, target_cols, drop = FALSE], na.rm = na.rm))
    } else {
      # Numeric input
      target_local <- match(target_indices_raw, original_indices)
      computed_scores <- scale(rowMeans(data[, target_local, drop = FALSE], na.rm = na.rm))
    }
  }
  
  # 3. Construct the Final Object
  #    This list holds EVERYTHING.
  results_object <- list(
    leaderboard = leaderboard,       # The table you see
    best_indices = best_indices,     # The numeric indices of the #1 set
    best_names = best_names,         # The character names of the #1 set
    scores = computed_scores,        # The calculated scores (if generate=TRUE)
    params = list(n.items = n.items, # Metadata
                  n.sets = n.sets)
  )
  
  # 4. Assign a Class
  #    This tells R to use our custom print function below
  class(results_object) <- "reduced_scale"
  
  return(results_object)
}

#' Print method for reduceTo results
#' @export
print.reduced_scale <- function(x, ...) {
  cat("\n--- Optimal Short-Form Scale Results ---\n")
  cat("Top", nrow(x$leaderboard), "Combinations:\n\n")
  
  # Print the dataframe nicely
  print(x$leaderboard)
  
  cat("\n----------------------------------------")
  if (!is.null(x$best_indices)) {
    cat("\nBest Items (Indices):", paste(x$best_indices, collapse = ", "))
  }
  if (!is.null(x$scores)) {
    cat("\nScores generated: Yes (Access via $scores)")
  }
  cat("\n")
}
